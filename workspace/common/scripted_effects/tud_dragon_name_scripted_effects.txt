            set_artifact_description = dragon_desc_monstrous_mature
        }
        else_if = {
            limit = { 
                has_artifact_feature = dragon_feature_build_monstrous
                has_artifact_modifier = ancient_dragon_modifier
            }
            set_artifact_description = dragon_desc_monstrous_ancient
        }
        else_if = {
            limit = { 
                has_artifact_feature = dragon_feature_build_monstrous
                has_artifact_modifier = dying_dragon_modifier
            }
            set_artifact_description = dragon_desc_monstrous_dying
        }
